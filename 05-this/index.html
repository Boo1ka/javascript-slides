this C++


class User {
public:
    int age=30;

    void showAge() {
        std::cout << this->age;
    };
};

int main() {
    User* sergey = new User;

    sergey->showAge(); // 30
}

this Java


public class User {
    private int age = 30;

    public void showAge() {
        System.out.println(this.age);
    }
}

public static void main(String []args){
    User sergey = new User();

    sergey.showAge();
}

this JavaScript


function User () {
    return {
        age: 30,

        showAge: function () {
            console.log(this.age);
        }
    }
}

var sergey = new User();

sergey.showAge();

Свойства this

  • ключевое слово
  • указывает на текущий объект
  • нельзя перезаписать
  • можно использовать за пределом объекта*

this за пределом объекта


this.innerWidth; // 300
        

this.process.version; // v7.0.0
